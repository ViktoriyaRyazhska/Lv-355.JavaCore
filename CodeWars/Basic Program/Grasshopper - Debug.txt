public class GrassHopper {

    public static String weatherInfo(int temp) {
        if (((temp - 32)*(5/9.0)) < 0) {
            return ((temp - 32)*(5/9.0) + " is freezing temperature");
            }
        else {
            return ((temp - 32)*(5/9.0)+ " is above freezing temperature");
      }
    }
}
